public void testDateWithOffset() throws Exception {
    ObjectMapper mapper = new ObjectMapper();
    mapper.setTimeZone(TimeZone.getTimeZone("GMT+2"));

    Date date = mapper.readValue("\"1970-01-01T00:00:00.000\"", java.util.Date.class);
    assertEquals("1970-01-01T00:00:00.000+02:00", ISODateTimeFormat.dateTime().print(date.getTime()));
}
