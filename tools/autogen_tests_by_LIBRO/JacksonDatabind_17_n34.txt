        .enableDefaultTyping()

        .enable(MapperFeature.REQUIRE\_SETTERS\_FOR\_GETTERS)
        .disable(MapperFeature.USE\_GETTERS\_AS\_SETTERS)

        .disable(SerializationFeature.FAIL\_ON\_EMPTY\_BEANS);
