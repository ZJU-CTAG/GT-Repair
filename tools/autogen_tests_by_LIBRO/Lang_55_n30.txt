public void testStopwatchSuspend() throws InterruptedException {
    StopWatch sw = new StopWatch();
    assertEquals(0, sw.getTime());
    assertEquals(0, sw.getSplitTime());
    assertEquals(0, sw.getLapTime());
    assertEquals(0, sw.getStartTime());
    assertEquals(0, sw.getTime());
    assertEquals(0, sw.getTime());
    sw.start();
    assertEquals(0, sw.getSplitTime());
    assertEquals(0, sw.getLapTime());
    assertTrue(sw.getStartTime() > 0);
    assertTrue(sw.getTime() > 0);
    Thread.sleep(1500);
    sw.suspend();
    long time1 = sw.getTime();
    System.out.println("time1 = " + time1);
    assertTrue(time1 > 0);
    Thread.sleep(2500);
    long time1a = sw.getTime();
    System.out.println("time1a = " + time1a);
