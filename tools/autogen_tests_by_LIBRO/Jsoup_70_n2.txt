public void testWhitespace() {
    String s = "<pre><code>\n"
            + "  message <span style=\"color:red\"> other   \n    message  with \n"
            + "   whitespaces      </span>\n"
            + "</code></pre>";
    Document doc = Jsoup.parse(s);
    assertEquals("<pre><code>\n"
            + "  message <span style=\"color:red\"> other   \n    message  with \n"
            + "   whitespaces      </span>\n"
            + "</code></pre>", doc.select("pre").first().outerHtml());
}
