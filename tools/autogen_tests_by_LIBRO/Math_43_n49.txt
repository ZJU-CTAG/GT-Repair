public void testIssue107() {
    int[] scores = {1, 2, 3, 4};
    SummaryStatistics stats = new SummaryStatistics();
    stats.setVarianceImpl(new Variance(false)); //use "population variance"
    for(int i : scores) {
      stats.addValue(i);
    }
    double sd = stats.getStandardDeviation();
    assertTrue(Double.isNaN(sd));
}
