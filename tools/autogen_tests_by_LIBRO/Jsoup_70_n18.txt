public void testPreserveWhitespacesInPreTag() throws IOException {
        String s = "<pre><code>\n"
                + "  message <span style=\"color:red\"> other   \n    message  with \n"
                + "   whitespaces      </span>\n"
                + "</code></pre>";
        Document doc = Jsoup.parse(s);
        System.out.println(doc.select("pre").first().html());
        assertEquals("  message <span style=\"color:red\"> other   \n    message  with \n   whitespaces </span>\n", doc.select("pre").first().html());
    }
