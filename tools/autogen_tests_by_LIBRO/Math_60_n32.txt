public void testCumulative() {
    NormalDistribution nd = new NormalDistributionImpl();
    for (int i = 0; i < 500; i++) {
        double val = Math.exp(i);
        try {
            System.out.println("val = " + val + " cumulative = " + nd.cumulativeProbability(val));
        } catch (MathException e) {
            e.printStackTrace();
            fail();
        }
    }
}
