public void testStopWatch() {
    StopWatch sw = new StopWatch();
    sw.start();
    Thread.sleep(1000);
    sw.suspend();
    System.out.println("Time 1: " + sw.getTime());
    Thread.sleep(2000);
    System.out.println("Time 1: " + sw.getTime());
    sw.resume();
    Thread.sleep(3000);
    sw.suspend();
    System.out.println("Time 2: " + sw.getTime());
    Thread.sleep(4000);
    System.out.println("Time 2: " + sw.getTime());
    Thread.sleep(5000);
    sw.stop();
    System.out.println("Time 3: " + sw.getTime());
}
