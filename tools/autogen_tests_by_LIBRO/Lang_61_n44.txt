public void testDeleteAll_String() {
    StrBuilder sb = new StrBuilder();
    sb.deleteAll("");
    assertEquals("", sb.toString());
    sb.append("abcdefghij");
    sb.deleteAll("");
    assertEquals("abcdefghij", sb.toString());
    sb.deleteAll("xyz");
    assertEquals("abcdefghij", sb.toString());
    sb.append("xxx");
    sb.deleteAll("xyz");
    assertEquals("abcdefghijxxx", sb.toString());
    sb.deleteAll("x");
    assertEquals("abcdefghij", sb.toString());
    sb.append("abcdefghij");
    sb.deleteAll("xyz");
    assertEquals("abcdefghijabcdefghij", sb.toString());
    sb.deleteAll("ab");
    assertEquals("cdefghijcdefghij", sb.toString());
    sb.deleteAll("cd");
