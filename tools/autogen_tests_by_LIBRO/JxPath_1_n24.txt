public void testSelfOrDescendant() throws Exception {
    final DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
    factory.setNamespaceAware(true);
    final DocumentBuilder builder = factory.newDocumentBuilder();
    final Document doc = builder.parse(new InputSource(new StringReader("<root id=\"1234\"/>")));

    final JXPathContext context = JXPathContext.newContext(doc);
    assertEquals("1234", context.selectSingleNode("//root/@id/text()"));
}
