public void testIssue107() {
    Document doc = Jsoup.parse("<html><head><style /></head><body><p>Whatever</p></body></html>");
    assertEquals("<html>\n" +
            " <head> \n" +
            "  <style></style> \n" +
            " </head> \n" +
            " <body> \n" +
            "  <p>Whatever</p> \n" +
            " </body>\n" +
            "</html>", doc.html());
}
