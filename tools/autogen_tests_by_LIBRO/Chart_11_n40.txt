public void testIssue868() {
    GeneralPath a = new GeneralPath();
    GeneralPath b = new GeneralPath();
    assertTrue(ShapeUtilities.equal(a,b));
    a.moveTo(0,0);
    b.moveTo(0,0);
    assertTrue(ShapeUtilities.equal(a,b));
    a.lineTo(0,0);
    b.lineTo(0,0);
    assertTrue(ShapeUtilities.equal(a,b));
    a.lineTo(1,0);
    b.lineTo(1,0);
    assertTrue(ShapeUtilities.equal(a,b));
    a.lineTo(1,1);
    b.lineTo(1,1);
    assertTrue(ShapeUtilities.equal(a,b));
    a.closePath();
    b.closePath();
    assertTrue(ShapeUtilities.equal(a,b));
    a.moveTo(1,1);
    b.moveTo(1,1);
    assertTrue(ShapeUtilities.equal(a,b));
