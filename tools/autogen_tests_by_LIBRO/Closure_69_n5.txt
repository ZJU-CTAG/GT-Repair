public void testIssue108() {
    goog.require('goog.graphics.Path');
    function demo() {
        var path = new goog.graphics.Path();
        var points = [[1,1], [2,2]];
        for (var i = 0; i < points.length; i++) {
            (i == 0 ? path.moveTo : path.lineTo)(points[i][0], points[i][1]);
        }
    }
    goog.exportSymbol('demo', demo);
}
