public void testDeleteAll_String() {
    StrBuilder sb = new StrBuilder();
    sb.deleteAll("");
    assertEquals("", sb.toString());
    sb.append("abcdefghij");
    sb.deleteAll("");
    assertEquals("abcdefghij", sb.toString());
    sb.append("abcdefghij");
    sb.deleteAll(null);
    assertEquals("abcdefghij", sb.toString());
    sb.append("abcdefghij");
    sb.deleteAll("gh");
    assertEquals("abcdefij", sb.toString());
    sb.append("abcdefghij");
    sb.deleteAll("ghi");
    assertEquals("abcdefj", sb.toString());
    sb.append("abcdefghij");
    sb.deleteAll("g");
    assertEquals("abcdefhij", sb.toString());
    sb.append("abcdefghij");
    sb.deleteAll("hi");
